CREATE TABLE `container_info_consumer` (
  `consumer_id` int NOT NULL AUTO_INCREMENT,
  `container_number` varchar(25) NOT NULL,
  `container_type` varchar(15) NOT NULL,
  `container_status` varchar(30) NOT NULL,
  `status_date_time` datetime NOT NULL,
  `current_location_code` varchar(10) NOT NULL,
  `vessel_name` varchar(100) NOT NULL,
  `voyage_number` varchar(50) NOT NULL,
  `port_of_loading` varchar(10) NOT NULL,
  `port_of_discharge` varchar(10) NOT NULL,
  `consumed_at` datetime DEFAULT CURRENT_TIMESTAMP,
  PRIMARY KEY (`consumer_id`)
)
